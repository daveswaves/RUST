# Make it executable
chmod +x rustup-init

./rustup-init

# Create rust project using cargo

cd to where you wish to create project

cargo new project_name

cd into project_name/ and run: cargo build.
cd target/debug
Run project: ./project_name

# cargo run is shorthand for the above.

'cargo check' identifies any issues without needing to compile/execute.

cd src
